packages = ["pandas", "plotly"] import pandas as pd import plotly.express as px import plotly.io as pio # Update the CSV file name df = pd.read_csv("state_exp_by_ports_and_countries.csv") states = sorted(df["State "].unique()) def get_top_20_by_state(state): subset = df[df["State "] == state] grouped = ( subset.groupby("Country", as_index=False) .agg({"AM24_INR": "sum", "AM24_USD": "sum"}) ) top_20 = grouped.sort_values("AM24_USD", ascending=False).head(20) table_html = top_20.to_html(index=False) fig = px.scatter_geo( top_20, locations="Country", locationmode="country names", hover_name="Country", size="AM24_USD", color="AM24_USD", projection="natural earth", title=f"Top 20 Destinations from {state}" ) map_html = pio.to_html(fig, full_html=False, include_plotlyjs="cdn") return table_html, map_html state_list = states
