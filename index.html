import asyncio import micropip async def main(): print("Python: Starting package installation...") # Install required packages. await micropip.install("pandas") await micropip.install("plotly") print("Python: Package installation complete.") import pandas as pd import plotly.express as px import plotly.io as pio from pyodide.http import pyfetch import io global df, state_list, get_top_20_by_state async def load_csv(): print("Python: Fetching CSV file...") response = await pyfetch("state_exp_by_ports_and_countries.csv") csv_text = await response.string() return pd.read_csv(io.StringIO(csv_text)) df = await load_csv() # Clean up column names. df.columns = df.columns.str.strip() state_list_local = sorted(df["State"].unique()) globals()["state_list"] = state_list_local # explicitly export the variable print("Python: CSV loaded successfully. DataFrame head:") print(df.head()) print("Python: Exported state_list:", state_list_local) def get_top_20_by_state(state): subset = df[df["State"] == state] grouped = subset.groupby("Country", as_index=False).agg({"AM24_INR": "sum", "AM24_USD": "sum"}) top_20 = grouped.sort_values("AM24_USD", ascending=False).head(20) table_html = top_20.to_html(index=False) fig = px.scatter_geo( top_20, locations="Country", locationmode="country names", hover_name="Country", size="AM24_USD", color="AM24_USD", projection="natural earth", title=f"Top 20 Destinations from {state}" ) map_html = pio.to_html(fig, full_html=False, include_plotlyjs="cdn") return table_html, map_html globals()["get_top_20_by_state"] = get_top_20_by_state print("Python: get_top_20_by_state function exported.") asyncio.ensure_future(main())
